public class edge {
  vertex startPoint;
  vertex endPoint;
  String name;

  public static int edgeNo = 1;
  public int currentNo;
  public edge(vertex a, vertex b){
    currentNo=edgeNo;
    startPoint = a;
    endPoint = b;
    name = "e" + "" + currentNo;
    edgeNo++;
  }
public String toStringUndirected(){
  return "(" + startPoint.currentName +"<->"+endPoint.currentName+")";
}
  @Override
  public String toString() {
    return "(" + startPoint.currentName +"->"+endPoint.currentName+")";
  }
}
